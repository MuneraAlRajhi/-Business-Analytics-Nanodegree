
 /* Query 1 */
SELECT G.Name, COUNT(T) AS Total_tracks
FROM Genre G
JOIN Track T
ON G.GenreId = T.GenreId
GROUP BY G.Name
ORDER BY 2 DESC
LIMIT 10;

/* Query 2 */
SELECT Al.title, SUM(I.total) AS Total_Invoice
FROM album Al
JOIN track T
ON Al.AlbumId = T.AlbumId
JOIN invoiceLine IL
ON T.TrackId = IL.TrackId
JOIN invoice I
ON IL.invoiceId = I.invoiceId
GROUP BY  Al.title
ORDER BY 2 DESC
LIMIT 5;

/* Query 3 */
SELECT C.country, SUM(I.Total) AS Total_Invoice
FROM Customer C
JOIN Invoice I
ON C.CustomerId = I.CustomerId
GROUP BY C.country
ORDER BY 2 DESC
LIMIT 10;

/* Query 4 */
SELECT T.name, count(PL.trackid) AS number_of_playlists
FROM Track T
JOIN playlisttrack PL
ON T.trackid = PL.trackid
GROUP BY T.name
ORDER BY 2 DESC
LIMIT 5;
